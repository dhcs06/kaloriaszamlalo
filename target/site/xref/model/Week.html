22  import javafx.collections.FXCollections;
23  import javafx.collections.ObservableList;
24  import javafx.scene.control.Alert;
25  import model.enumerators.EWeekDayTypes;
26  import model.interfaces.IWeekLogic;
27  
28  import java.util.*;
29  
30  /**
31   * Represents the days of the week.
32   */
33  public class Week implements IWeekLogic {
34  	/**
35  	 * Adds the meals to the days of the week.
36  	 */
37  	private final HashMap<EWeekDayTypes, List<Food>> daysWithFoods;
38  
39  	/**
40  	 * Constructor of the Week's class.
41  	 */
42  	public Week() {
43  		this.daysWithFoods = new HashMap<>();
44  	}
45  
46  	/**
47  	 * Overriding of the Week's constructor.
48  	 *
49  	 * @param days A hashmap dictionary which contains the days
50  	 */
51  	public Week(HashMap<EWeekDayTypes, List<Food>> days) {
52  		this.daysWithFoods = days;
53  	}
54  
55  	/**
56  	 * This function add foodlists to a specific date.
57  	 *
58  	 * @param day      The day where we want to add the list of foods
59  	 * @param foodList The list of foods what we want to add
60  	 */
61  	public void AddFoodsToDay(EWeekDayTypes day, List<Food> foodList) {
62  		this.daysWithFoods.put(day, foodList);
63  	}
64  
65  	/**
66  	 * Returns the list of foods according to the day.
67  	 *
68  	 * @param day The day itself
69  	 * @return list of foods
70  	 */
71  	public List<Food> GetFoodsFromDay(EWeekDayTypes day) {
72  		return this.daysWithFoods.get(day);
73  	}
74  
75  	/**
76  	 * This function generates a list of Foods according to nuOfCalories.
77  	 *
78  	 * @param listOfSelectableFoods The list of foods what generated
79  	 * @param nuOfCalories          The number of calories
80  	 */
81  	@Override
82  	public void GenerateAWeekWithFoods(List<Food> listOfSelectableFoods, int nuOfCalories) {
83  
84  		int min = 0;
85  		int max = listOfSelectableFoods.size() - 1;
86  
87  		Random r = new Random();
88  
89  		for (EWeekDayTypes day : EWeekDayTypes.values()) {
90  			ObservableList<Food> listOfFoods = FXCollections.observableArrayList();
91  			int consumedCalories = 0;
92  			while (consumedCalories < nuOfCalories) {
93  				int randomIndexForListOfSelectableFoods = r.nextInt(max - min + 1) + min;
94  				Food food = listOfSelectableFoods.get(randomIndexForListOfSelectableFoods);
95  				consumedCalories += food.GetCalorie();
96  				if (consumedCalories < nuOfCalories) {
97  					listOfFoods.add(food);
98  				}
99  			}
100 			this.AddFoodsToDay(day, listOfFoods);
101 		}
102 	}
103 
104 	/**
105 	 * This function calculates the bmi value according to the height and weight.
106 	 *
107 	 * @param height The height of the user
108 	 * @param weight The weight of the user
109 	 * @return the calculated value from the weight and height
110 	 *
111 	 */
112 	@Override
113 	public float calculateTheBmiValueAccording(float height, float weight) {
114 		if (height == 0) return 0;
115 		return (weight / (height * height)) * 10000;
116 	}
117 
118 	/**
119 	 * This function calculates the calories value according to the bmi's value.
120 	 *
121 	 * @param bmiValue The value we calculate from the user's weight and height
122 	 * @return return with the kCal value the user should eat
123 	 */
124 	@Override
125 	public int calculateTheCaloriesValueAccording(float bmiValue) {
126 		Alert alert2 = new Alert(Alert.AlertType.INFORMATION);
127 		alert2.setTitle("Ups...");
128 		alert2.setHeaderText("Information");
129 		alert2.setContentText("Wrong data!");
130 
131 
132 		if (bmiValue <= 10 || bmiValue >= 40) {
133 			alert2.show();
134 		} else {
135 
136 			if (bmiValue > 11 && bmiValue <= 18.5) {
137 				return 2500;
138 			} else if (bmiValue >= 18.6 && bmiValue < 25) {
139 				return 2000;
140 			} else if (bmiValue >= 25 && bmiValue < 30) {
141 				return 1500;
142 			} else if (bmiValue >= 30 && bmiValue < 40) {
143 				return 1000;
144 			}
145 		}
146 		return 0;
147 	}
148 }
